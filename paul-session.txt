│ Plan to implement                                                            │
│                                                                              │
│ Web Hosting Ticketing System                                                 │
│                                                                              │
│ Context                                                                      │
│                                                                              │
│ A web hosting company needs a customer-facing support ticketing system.      │
│ Customers submit tickets when their websites have problems. Staff view and   │
│ manage tickets. Starting from an empty directory, using Python + Flask +     │
│ SQLite.                                                                      │
│                                                                              │
│ File Structure                                                               │
│                                                                              │
│ hackathon/                                                                   │
│ ├── app.py               # Flask app, all routes, DB helpers, form           │
│ validation                                                                   │
│ ├── schema.sql           # SQLite DDL (CREATE TABLE IF NOT EXISTS)           │
│ ├── requirements.txt     # flask>=3.0                                        │
│ ├── tickets.db           # Created at runtime by init_db()                   │
│ ├── static/                                                                  │
│ │   └── style.css        # Layout, forms, severity color badges, table       │
│ styles                                                                       │
│ └── templates/                                                               │
│     ├── base.html        # Shared layout: nav, flash messages, footer        │
│     ├── submit.html      # Customer form (all 6 fields, per-field error      │
│ display)                                                                     │
│     ├── submit_success.html  # Ticket confirmation with ID                   │
│     ├── ticket_list.html # Paginated staff view with severity filter         │
│     └── ticket_detail.html  # Full ticket view for staff                     │
│                                                                              │
│ Database Schema (schema.sql)                                                 │
│                                                                              │
│ CREATE TABLE IF NOT EXISTS tickets (                                         │
│     id            INTEGER  PRIMARY KEY AUTOINCREMENT,                        │
│     customer_name TEXT     NOT NULL,                                         │
│     email         TEXT     NOT NULL,                                         │
│     url           TEXT     NOT NULL,                                         │
│     severity      TEXT     NOT NULL CHECK(severity IN                        │
│ ('Low','Medium','High','Critical')),                                         │
│     problem_time  TEXT     NOT NULL,   -- ISO-8601: "YYYY-MM-DDTHH:MM"       │
│     description   TEXT     NOT NULL,                                         │
│     submitted_at  TEXT     NOT NULL DEFAULT                                  │
│ (strftime('%Y-%m-%dT%H:%M:%S','now'))                                        │
│ );                                                                           │
│                                                                              │
│ Routes                                                                       │
│                                                                              │
│ ┌──────────┬──────────────────────────┬────────────────────────────────────┐ │
│ │  Method  │           Path           │              Purpose               │ │
│ ├──────────┼──────────────────────────┼────────────────────────────────────┤ │
│ │ GET/POST │ /                        │ Submit ticket form                 │ │
│ │          │                          │ (customer-facing)                  │ │
│ ├──────────┼──────────────────────────┼────────────────────────────────────┤ │
│ │ GET      │ /success/<int:ticket_id> │ Confirmation page after submission │ │
│ ├──────────┼──────────────────────────┼────────────────────────────────────┤ │
│ │ GET      │ /tickets                 │ Paginated ticket list with         │ │
│ │          │                          │ optional ?severity= filter (staff) │ │
│ ├──────────┼──────────────────────────┼────────────────────────────────────┤ │
│ │ GET      │ /tickets/<int:ticket_id> │ Full ticket detail (staff)         │ │
│ └──────────┴──────────────────────────┴────────────────────────────────────┘ │
│                                                                              │
│ Key Implementation Details                                                   │
│                                                                              │
│ app.py                                                                       │
│                                                                              │
│ - get_db() / close_db() using Flask g and app.teardown_appcontext            │
│ - sqlite3.Row as row_factory for name-based column access in templates       │
│ - init_db() reads and executes schema.sql via app.open_resource(); called at │
│  startup                                                                     │
│ - SEVERITY_LEVELS = ["Low", "Medium", "High", "Critical"]                    │
│ - POST / uses POST-Redirect-GET pattern: on success redirect to              │
│ /success/<id> (prevents duplicate submissions on refresh)                    │
│ - On validation failure: re-render form passing errors dict and              │
│ form_data=request.form (preserves user input)                                │
│                                                                              │
│ Server-side Validation (in POST handler)                                     │
│                                                                              │
│ ┌───────────────┬──────────────────────────────────────────────────────────┐ │
│ │     Field     │                          Rules                           │ │
│ ├───────────────┼──────────────────────────────────────────────────────────┤ │
│ │ customer_name │ Required, strip, max 120 chars                           │ │
│ ├───────────────┼──────────────────────────────────────────────────────────┤ │
│ │ email         │ Required, must contain @ and ., max 200 chars            │ │
│ ├───────────────┼──────────────────────────────────────────────────────────┤ │
│ │ url           │ Required, strip, max 500 chars                           │ │
│ ├───────────────┼──────────────────────────────────────────────────────────┤ │
│ │ severity      │ Must be in SEVERITY_LEVELS                               │ │
│ ├───────────────┼──────────────────────────────────────────────────────────┤ │
│ │ problem_time  │ Required, must parse as %Y-%m-%dT%H:%M via               │ │
│ │               │ datetime.strptime                                        │ │
│ ├───────────────┼──────────────────────────────────────────────────────────┤ │
│ │ description   │ Required, strip, max 5000 chars                          │ │
│ └───────────────┴──────────────────────────────────────────────────────────┘ │
│                                                                              │
│ Pagination (/tickets)                                                        │
│                                                                              │
│ - 20 tickets per page, ?page=N query param                                   │
│ - Optional ?severity= filter passed to SQL WHERE clause                      │
│ - Pass page, total_pages, severity_filter to template                        │
│                                                                              │
│ CSS (static/style.css)                                                       │
│                                                                              │
│ - Centered layout, max-width 900px                                           │
│ - Full-width form inputs/selects/textareas                                   │
│ - .has-error red border + .error red text for field errors                   │
│ - Severity badge colors: Low=green, Medium=yellow/amber, High=orange,        │
│ Critical=red                                                                 │
│ - Alternating table row colors for ticket list                               │
│ - Flash message alert banner                                                 │
│                                                                              │
│ Verification                                                                 │
│                                                                              │
